BeanеңЁSpringдёӯжҳҜеҰӮдҪ•иЎЁзӨәзҡ„пјҹ

#жәҗз Ғ

1. иғҢжҷҜ

дёәд»Җд№ҲиҰҒжҗһжё…жҘҡBeanеңЁSpringдёӯжҳҜеҰӮдҪ•иЎЁзӨәзҡ„пјҢдё»иҰҒжңүд»ҘдёӢдёӨдёӘж–№йқўзҡ„еҺҹеӣ пјҡ

  1. еҜ№дәҺдҪҝз”ЁиҖ…жқҘиҜҙпјҢеңЁдҪҝз”ЁXMLжҲ–жіЁи§Јй…ҚзҪ®Beanж—¶пјҢжҲ‘д»¬йңҖиҰҒзҹҘйҒ“жңүе“ӘдәӣеҸҜй…ҚзҪ®йЎ№пјҢжүҚиғҪиҝӣиЎҢй…ҚзҪ®пјӣ
  2. еҜ№дәҺејҖеҸ‘иҖ…жқҘиҜҙпјҢеңЁSpringеӨ„зҗҶBeanзҡ„иҝҮзЁӢдёӯпјҢдјҡдҫқиө–BeanеңЁSpringдёӯзҡ„иЎЁзӨәпјҢжҳҜйҳ…иҜ»жәҗз Ғзҡ„еүҚзҪ®зҹҘиҜҶгҖӮ

SpringзӣёеҪ“дәҺдёҖдёӘз”»з¬”пјҢдҪ еҸҜд»ҘйҖҡиҝҮи°ғиҠӮз”»з¬”зҡ„еҗ„йЎ№еҸӮж•°пјҲBeanзҡ„й…ҚзҪ®йЎ№пјүпјҢеҰӮз¬”зҡ„зІ—з»ҶпјҢйўңиүІзӯүпјҢжқҘеё®жҲ‘д»¬з”»еҮәеӨҡз§ҚеӨҡж ·зҡ„з”»пјҲBeanпјүгҖӮйӮЈе°ұи®©жҲ‘д»¬дёҖиө·зңӢзңӢиҝҷдёӘз”»з¬”жңүе“ӘдәӣеҸӮж•°пјҹ

2. еҰӮжһңи®©дҪ и®ҫи®ЎпјҢдҪ дјҡеҰӮдҪ•и®ҫи®Ўпјҹ

жҲ‘д»¬дҪҝз”ЁSpringжҳҜдёәдәҶи®©Springеё®жҲ‘д»¬еҲӣе»әе’Ңз®ЎзҗҶдҫқиө–пјҲд№ҹе°ұжҳҜBeanпјүгҖӮйӮЈеҲӣе»әдёҖдёӘдҫқиө–йңҖиҰҒе“Әдәӣй…ҚзҪ®йЎ№е‘ўпјҹдёҚйҡҫжғіеҲ°д»ҘдёӢиҝҷдәӣй…ҚзҪ®йЎ№пјҡ

  • зұ»зҡ„еҗҚз§°
  • еұһжҖ§зҡ„еҖјпјҢеҰӮжһңжңүзҡ„иҜқ
  • жһ„йҖ еҷЁеҸӮж•°зҡ„еҖјпјҢеҰӮжһңжңүзҡ„иҜқ

еҰӮжһңиҜҙжҲ‘д»¬иҝҳйңҖиҰҒSpringеё®жҲ‘д»¬еҒҡдёҖдәӣйўқеӨ–еҠЁдҪңпјҢеҰӮеңЁBeanеҲӣе»әжҲ–й”ҖжҜҒж—¶еҲҶеҲ«жү§иЎҢжҹҗдёӘзү№е®ҡж–№жі•пјҢжҲ‘д»¬еҸҜд»Ҙж·»еҠ д»ҘдёӢиҝҷдәӣй…ҚзҪ®йЎ№пјҡ

  • еҲқе§ӢеҢ–ж–№жі•
  • й”ҖжҜҒж–№жі•

еҰӮжһңиҜҙжҲ‘д»¬е№¶дёҚд»…йҷҗдәҺдҪҝз”Ёжһ„йҖ еҷЁжҲ–еұһжҖ§зҡ„ж–№ејҸеҺ»еҲӣе»әдёҖдёӘдҫқиө–пјҢиҖҢжҳҜйҖҡиҝҮе·ҘеҺӮзҡ„ж–№ејҸеҺ»еҲӣе»әдҫқиө–пјҢжҲ‘д»¬еҸҜд»Ҙж·»еҠ д»ҘдёӢиҝҷдәӣй…ҚзҪ®йЎ№пјҡ

  • factory-beanпјҢз”ЁдәҺжҢҮе®ҡе·ҘеҺӮ
  • factory-methodпјҢз”ЁдәҺжҢҮе®ҡе·ҘеҺӮдёӯеҲӣе»әBeanзҡ„ж–№жі•
  • вҖҰ

дёҚйҡҫеҸ‘зҺ°пјҢеҜ№дәҺејҖеҸ‘иҖ…жқҘиҜҙпјҢеҪ“жҲ‘д»¬йңҖиҰҒи¶ҠеӨҡзҡ„еҠҹиғҪпјҢд№ҹе°ұйңҖиҰҒжӣҙеӨҡзҡ„й…ҚзҪ®йЎ№пјҢеҜ№Beanзҡ„иЎЁзӨәд№ҹе°ұи¶ҠеӨҚжқӮгҖӮиҖҢеҜ№дәҺдҪҝз”ЁиҖ…жқҘиҜҙпјҢд№ҹе°ұжңүдәҶжӣҙеӨҡеҸҜй…ҚзҪ®д№ҹе°ұжҳҜеҸҜиҮӘе®ҡд№үеҠҹиғҪзҡ„ең°ж–№гҖӮ

SpringдёәдәҶжӣҙеҘҪжӣҙе…Ёйқўең°ж»Ўи¶із”ЁжҲ·зҡ„йңҖжұӮпјҢж·»еҠ дәҶи®ёеӨҡдёҚеёёз”Ёзҡ„еҠҹиғҪпјҢиҝҷж ·зҡ„еҘҪеӨ„жҳҜеӨҚжқӮеәҰиҪ¬з§»з»ҷдәҶSpringзҡ„ејҖеҸ‘иҖ…пјҢиҖҢдҪҝз”ЁиҖ…еҸӘйңҖиҰҒе…іжіЁиҮӘе·ұзҡ„дёҡеҠЎпјҢз„¶еҗҺж №жҚ®дёҡеҠЎеҒҡзӣёеә”зҡ„й…ҚзҪ®еҚіеҸҜпјҢиҝҷд№ҹжҳҜдёҖдёӘжҲҗзҶҹзҡ„дјҒдёҡзә§жЎҶжһ¶зҡ„и®ҫи®ЎжҖқжғігҖӮ

дёәд»Җд№Ҳдјҡжңүи¶ҠжқҘи¶ҠеӨҡзҡ„й…ҚзҪ®йЎ№пјҹжӢҝжүӢжңәзҡ„и®ҫзҪ®дёәдҫӢпјҢиө·еҲқеҸҜиғҪеҸӘиғҪз®Җз®ҖеҚ•еҚ•зҡ„и®ҫзҪ®дёӢеЈҒзәёпјҢи“қзүҷзӯүпјҢйҡҸзқҖйңҖжұӮзҡ„еўһеҠ пјҢзҺ°еңЁзҡ„жүӢжңәеҸҜд»Ҙи®ҫзҪ®WIFIпјҢеә”з”Ёзҡ„йҖҡзҹҘзұ»еһӢзӯүпјҢжң¬иҙЁдёҠе°ұжҳҜдёәз”ЁжҲ·жҸҗдҫӣжӣҙе®ҡеҲ¶еҢ–зҡ„еҠҹиғҪгҖӮиҜ•жғіеҰӮжһңдҪ зҡ„жүӢжңәи®ҫзҪ®дёҚж”ҜжҢҒи®ҫзҪ®WIFIпјҢйӮЈжҳҜеӨҡд№ҲжӮІжғЁзҡ„дёҖд»¶дәӢгҖӮ

3. SpringжҳҜеҰӮдҪ•и®ҫи®ЎBeanзҡ„иЎЁзӨәзҡ„пјҹ

еҗҢжҲ‘д»¬дёҠйқўзҡ„и®ҫи®Ўзұ»дјјпјҢSpringеңЁжҲ‘д»¬и®ҫи®Ўзҡ„еҹәзЎҖдёҠпјҢиҝҳз»ҙжҠӨдәҶеҰӮжҳҜеҗҰе»¶иҝҹеҲқе§ӢеҢ–гҖҒдҪңз”ЁеҹҹгҖҒжүҖйңҖдҫқиө–зӯүй…ҚзҪ®йЎ№гҖӮдҪ“зҺ°еңЁжәҗз ҒдёҠдҫҝжҳҜBeanDefinitionжҺҘеҸЈдёӯеЈ°жҳҺзҡ„еҗ„дёӘиҺ·еҸ–ж–№жі•пјҢеҰӮдёӢзұ»еӣҫпјҡ

д»Һз»§жүҝе…ізі»дёҠжқҘзңӢпјҢBeanDefinitionз»§жүҝдәҶд»ҘдёӢжҺҘеҸЈпјҡ

  • BeanMetadataElementжҺҘеҸЈпјҢз”ЁдәҺиҝ”еӣһBeanзҡ„е…ғж•°жҚ®жқҘжәҗдәҺе“ӘйҮҢпјҢд№ҹе°ұжҳҜиҝҷдёӘBeanзҡ„й…ҚзҪ®дҝЎжҒҜжңҖеҲқжҳҜеңЁе“ӘйҮҢи®ҫзҪ®зҡ„пјҢжҜ”еҰӮжҳҜеңЁXMLдёӯпјҢиҝҳжҳҜеңЁй…ҚзҪ®зұ»дёӯгҖӮ

  • AttributeAccessorжҺҘеҸЈпјҢз”ЁдәҺи®°еҪ•дёҖдәӣеҶ…йғЁзҡ„еұһжҖ§д»ҘдҫӣеҗҺз»ӯSpringеӨ„зҗҶBeanDefinitionзҡ„иҝҮзЁӢдёӯдҪҝз”ЁгҖӮ

д»Һе®һзҺ°дёҠжқҘзңӢпјҢBeanDefinitionжңүд»ҘдёӢеӯҗзұ»пјҡ

  • AbstractBeanDefinitionпјҢе…¶жҳҜBeanDefinitionзҡ„йҖҡз”ЁжҠҪиұЎе®һзҺ°пјҢд№ҹжҳҜжүҖжңүBeanDefinitionе®һзҺ°зұ»зҡ„зҲ¶зұ»
  • еҪ“Springе®№еҷЁеҶ…йғЁеӨ„зҗҶBeanDefinitionзҡ„ж—¶еҖҷдјҡдҪҝз”ЁRootBeanDefinition
  • еҪ“дҪҝз”ЁXMLзҡ„ж–№ејҸжіЁеҶҢBeanDefinitionзҡ„ж—¶еҖҷдјҡдҪҝз”ЁGenericBeanDefinition
  • еҪ“дҪҝз”Ёз»„д»¶жү«жҸҸзҡ„ж–№ејҸжіЁеҶҢBeanDefinitionзҡ„ж—¶еҖҷдјҡдҪҝз”ЁScannedGenericBeanDefinition
  • еҪ“дҪҝз”Ёзј–зЁӢејҸжіЁеҶҢзҡ„ж–№ејҸжіЁеҶҢBeanDefinitionзҡ„ж—¶еҖҷдјҡдҪҝз”ЁAnnotatedGenericBeanDefinition
  • еҪ“дҪҝз”Ёй…ҚзҪ®зұ»зҡ„@Beanзҡ„ж–№ејҸжіЁеҶҢBeanDefinitionзҡ„ж—¶еҖҷдјҡдҪҝз”ЁConfigurationClassBeanDefinition

4. дёәд»Җд№ҲиҰҒжӢҶеҲҶдёәиҝҷд№ҲеӨҡзҡ„BeanDefinitionпјҹ

д»ҺйңҖжұӮдёҠжқҘзңӢпјҢиө·еҲқSpringеҸӘиғҪйҖҡиҝҮXMLжіЁеҶҢBeanDefinitionпјҢдҪҝз”Ёзҡ„жҳҜRootBeanDefinitionзҡ„з»§жүҝдҪ“зі»пјӣеҗҺжқҘдёәдәҶиғҪеӨҹйҖҡиҝҮжіЁи§ЈжіЁеҶҢBeanDefinitionпјҢеңЁSpring2.5еј•е…ҘдәҶGenericBeanDefinitionзҡ„з»§жүҝдҪ“зі»гҖӮ
д»Һи®ҫи®ЎдёҠжқҘзңӢпјҢBeanDefinitionзҡ„дёҚеҗҢзҡ„еӯҗзұ»йҖӮз”ЁдәҺдёҚеҗҢзҡ„еңәжҷҜпјҢжҜ”еҰӮеҪ“дҪҝз”ЁжіЁи§Јзҡ„ж–№ејҸжіЁеҶҢBeanDefinitionж—¶пјҢйңҖиҰҒжҗәеёҰжүҖж ҮжіЁжіЁи§Јзҡ„дҝЎжҒҜпјҲд№ҹе°ұжҳҜAnnotationMetadataпјүпјҢиҮӘз„¶иҰҒеј•е…Ҙж–°зҡ„е®һзҺ°зұ»гҖӮ

жұҪиҪҰпјҢе…¬дәӨиҪҰпјҢиҙ§иҪҰиҷҪз„¶йғҪжҳҜиҪҰпјҢдҪҶжңүзқҖдёҚеҗҢзҡ„йҖӮз”ЁеңәжҷҜгҖӮжғіиұЎдёҖдёӢдёҖдёӘиҙ§иҪҰжӢүдәҶдёҖж»ЎиҪҰзҡ„дәәпјҢд№ҹиғҪжӢүпјҢе°ұжҳҜдёҚеӨӘеҗҲйҖӮгҖӮ

5. жәҗз Ғеү–жһҗ-еҰӮдҪ•иЎЁзӨәжһ„йҖ еҷЁеҸӮж•°пјҹ

д»Һи®ҫи®ЎдёҠжқҘзңӢпјҢжһ„йҖ еҷЁеҸӮж•°ж— йқһе°ұеҸӮж•°зҡ„дёӢж ҮпјҢеҸҠеҸӮж•°зҡ„еҖјдёӨйғЁеҲҶгҖӮ

жҲ‘д»¬еҸҜд»ҘйҖҡиҝҮз»ҙжҠӨдёҖдёӘMapжқҘз»ҙжҠӨеӨҡдёӘжһ„йҖ еҷЁеҸӮж•°пјҢй”®е°ұд»ЈиЎЁеҸӮж•°зҡ„дёӢж ҮпјҢеҖје°ұд»ЈиЎЁеҸӮж•°зҡ„еҖјпјӣеҸҰеӨ–иҝҳйңҖиҰҒйўқеӨ–ж·»еҠ дёҖдёӘListжқҘз»ҙжҠӨжіӣеһӢеҸӮж•°гҖӮ

ConstructorArgumentValues#ConstructorArgumentValues(ConstructorArgumentValues) дҪҝз”ЁдәҶеҺҹеһӢжЁЎејҸпјҢжҸҗдҫӣдәҶеҗҢзұ»еһӢжһ„йҖ еҷЁд»Ҙж·ұжӢ·иҙқеҜ№иұЎгҖӮ

6. жәҗз Ғеү–жһҗ-еҰӮдҪ•иЎЁзӨәеұһжҖ§еҖјпјҹ

д»Һи®ҫи®ЎдёҠжқҘзңӢпјҢеұһжҖ§еҖјж— йқһе°ұеұһжҖ§зҡ„еҗҚз§°пјҢеҸҠеұһжҖ§зҡ„еҖјдёӨйғЁеҲҶгҖӮ

жҲ‘д»¬еҸҜд»ҘйҖҡиҝҮPropertyValueзұ»жқҘд»ЈиЎЁдёҖдёӘеұһжҖ§еҖјпјҢе…¶еҢ…еҗ«дәҶеұһжҖ§зҡ„еҗҚз§°пјҢеұһжҖ§зҡ„еҖјпјҢиҪ¬жҚўеҗҺзҡ„еҖјпјҢд»ҘеҸҠжҳҜеҗҰе·Іиў«иҪ¬жҚўзӯүеӯ—ж®өгҖӮ

дёәдәҶж–№дҫҝз»ҙжҠӨеӨҡдёӘеұһжҖ§еҖјпјҢжҲ‘д»¬еҸҜд»ҘйҖҡиҝҮMutablePropertyValuesжқҘжҢҒжңүдёҖдёӘPropertyValueзҡ„еҲ—иЎЁгҖӮдёәдәҶж–№дҫҝеҜ№еұһжҖ§еҖјзҡ„йҒҚеҺҶпјҢжҲ‘д»¬еҸҜд»Ҙе®һзҺ°дәҶPropertyValuesжҺҘеҸЈпјҢжң¬иҙЁд№ҹжҳҜдёҖдёӘPropertyValueзҡ„иҝӯд»ЈеҷЁгҖӮ

MutablePropertyValuesдҪҝз”ЁдәҶиҝӯд»ЈеҷЁжЁЎејҸпјҢж–№дҫҝеҜ№еӨҡдёӘеұһжҖ§еҖјиҝӣиЎҢйҒҚеҺҶпјӣе№¶з»ҙжҠӨдәҶеӨҡдёӘеұһжҖ§еҖјдёәе…¶жҸҗдҫӣз»ҹдёҖзҡ„и®ҝй—®ж“ҚдҪңпјҢд№ҹжҳҜдёҖз§Қз»„еҗҲжЁЎејҸгҖӮ

7. жәҗз Ғеү–жһҗ-еҰӮдҪ•е®һзҺ°йҖҡз”Ёзҡ„BeanDefinitionпјҹ

BeanDefinitionзҡ„йҖҡз”ЁжҠҪиұЎе®һзҺ°жҳҜAbstractBeanDefinitionпјҢд№ҹжҳҜжүҖжңүBeanDefinitionе®һзҺ°зұ»зҡ„зҲ¶зұ»пјҢз»ҙжҠӨдәҶBeanе®ҡд№үзӣёе…ізҡ„й…ҚзҪ®дҝЎжҒҜпјҢд№ҹжҸҗдҫӣдәҶеҜ№е…¶иҝӣиЎҢи®ҫзҪ®е’ҢиҺ·еҸ–зҡ„ж–№жі•гҖӮ

import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.config.BeanDefinition;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

@Slf4j
public class BeanDefinitionTypeTest {

    public static void main(String[] args) {
        testGenericBeanDefinition();
        testAnnotatedGenericBeanDefinition();
        testScannedGenericBeanDefinition();
        testConfigurationClassBeanDefinition();
    }

    public static void testGenericBeanDefinition() {
        // дҪҝз”ЁXMLзҡ„ж–№ејҸ
        ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext(BeanDefinitionTypeTest.class.getPackage().getName().replace(".", "/") + "/beans.xml");
        context.refresh();

        final BeanDefinition beanDefinition = context.getBeanFactory().getBeanDefinition("testBean");
        final TestBean bean = context.getBean(TestBean.class);
        printBeanInfo("XML", beanDefinition, bean);
    }

    public static void testAnnotatedGenericBeanDefinition() {
        // дҪҝз”Ёзј–зЁӢејҸжіЁеҶҢзҡ„ж–№ејҸ
        AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();
        context.register(TestComponent.class);
        context.refresh();

        final BeanDefinition beanDefinition = context.getBeanFactory().getBeanDefinition("testComponent");
        final TestComponent bean = context.getBean(TestComponent.class);
        printBeanInfo("Programming", beanDefinition, bean);
    }

    public static void testScannedGenericBeanDefinition() {
        // дҪҝз”Ёз»„д»¶жү«жҸҸзҡ„ж–№ејҸ
        AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();
        context.scan(BeanDefinitionTypeTest.class.getPackage().getName());
        context.refresh();
        final BeanDefinition beanDefinition = context.getBeanFactory().getBeanDefinition("testComponent");
        final TestComponent bean = context.getBean(TestComponent.class);
        printBeanInfo("Scan", beanDefinition, bean);
    }


    public static void testConfigurationClassBeanDefinition() {
        // дҪҝз”Ёй…ҚзҪ®зұ»зҡ„@Beanзҡ„ж–№ејҸ
        AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();
        context.register(TestConfiguration.class);
        context.refresh();

        final BeanDefinition beanDefinition = context.getBeanFactory().getBeanDefinition("testBean");
        final TestBean bean = context.getBean(TestBean.class);
        printBeanInfo("@Bean", beanDefinition, bean);
    }

    private static void printBeanInfo(String from, BeanDefinition beanDefinition, Object bean) {
        log.info("Loaded from: {}, bean definition type: {}, bean definition: {}, bean: {}", from, beanDefinition.getClass(), beanDefinition, bean);
    }
}

import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.config.BeanDefinition;
import org.springframework.beans.factory.config.ConfigurableBeanFactory;
import org.springframework.beans.factory.support.BeanDefinitionBuilder;
import org.springframework.beans.factory.support.DefaultListableBeanFactory;

@Slf4j
public class BeanDefinitionBuilderTest {
    public static void main(String[] args) {
        testBeanDefinitionBuilder();
    }

    public static void testBeanDefinitionBuilder() {
        // дҪҝз”Ё Builder жЁЎејҸзј–зЁӢејҸеЈ°жҳҺдёҖдёӘ BeanDefinition
        final BeanDefinition beanDefinition = BeanDefinitionBuilder
                .genericBeanDefinition(TestBean.class)
                .addPropertyValue("name", "foo")
                .addPropertyValue("age", 1)
                .setScope(ConfigurableBeanFactory.SCOPE_SINGLETON)
                .setInitMethodName("initialize")
                .getBeanDefinition();

        final DefaultListableBeanFactory beanFactory = new DefaultListableBeanFactory();
        // жіЁеҶҢ BeanDefinition
        beanFactory.registerBeanDefinition("testBean", beanDefinition);

        // иҺ·еҸ– Bean
        final Object bean = beanFactory.getBean("testBean");
        printBeanInfo("BeanDefinitionBuilder", beanDefinition, bean);
    }

    private static void printBeanInfo(String from, BeanDefinition beanDefinition, Object bean) {
        log.info("Loaded from: {}, bean definition type: {}, bean definition: {}, bean: {}", from, beanDefinition.getClass(), beanDefinition, bean);
    }
}
